========================================================
  Конфигурация License Server — описание параметров
========================================================

Все файлы читаются через configparser (формат INI).
Переменные окружения имеют приоритет над значениями в .cfg.

--------------------------------------------------------
  app.cfg — основные параметры приложения
--------------------------------------------------------

[app]

name
  Отображаемое название приложения.
  Тип: строка
  Пример: name = My License Server

secret_key
  Секретный ключ для подписи JWT-токенов администратора.
  ОБЯЗАТЕЛЬНО смените перед деплоем!
  Переменная окружения: SECRET_KEY
  Тип: строка (рекомендуется минимум 32 случайных символа)
  Пример: secret_key = a8f3k2...

api_secret
  Секрет для подписи запросов публичного API (HMAC).
  Переменная окружения: API_SECRET
  Тип: строка

jwt_algorithm
  Алгоритм подписи JWT.
  Допустимые значения: HS256, HS384, HS512
  По умолчанию: HS256

token_expires_hours
  Время жизни JWT-токена администратора в часах.
  Тип: целое число > 0
  По умолчанию: 8
  Пример: token_expires_hours = 24

debug
  Включает отладочный режим FastAPI (подробные ошибки, автоперезагрузка).
  НЕ включайте на продакшене!
  Допустимые значения: true / false
  По умолчанию: false

--------------------------------------------------------
  database.cfg — подключение к базе данных
--------------------------------------------------------

[database]

url
  URL подключения к БД в формате SQLAlchemy async.
  Переменная окружения: DATABASE_URL
  Примеры:
    SQLite:     sqlite+aiosqlite:///./data/licserver.db
    PostgreSQL: postgresql+asyncpg://user:pass@localhost:5432/licserver

echo_sql
  Выводить все SQL-запросы в лог (только для отладки).
  Допустимые значения: true / false
  По умолчанию: false

pool_size
  Размер пула соединений (не применяется для SQLite).
  Тип: целое число
  По умолчанию: 5

max_overflow
  Максимальное число соединений сверх pool_size.
  Тип: целое число
  По умолчанию: 10

--------------------------------------------------------
  smtp.cfg — отправка e-mail
--------------------------------------------------------

[smtp]

host
  Адрес SMTP-сервера.
  Пример: smtp.gmail.com, smtp.yandex.ru

port
  Порт SMTP-сервера.
  Типичные значения: 587 (STARTTLS), 465 (SSL), 25
  По умолчанию: 587

user
  Логин для аутентификации на SMTP-сервере.
  Переменная окружения: SMTP_USER

password
  Пароль для аутентификации на SMTP-сервере.
  Переменная окружения: SMTP_PASSWORD

from
  Адрес отправителя в письмах.
  Пример: noreply@example.com

tls
  Использовать STARTTLS при подключении.
  Допустимые значения: true / false
  По умолчанию: true

enabled
  Включить отправку e-mail. При false письма не отправляются.
  Допустимые значения: true / false
  По умолчанию: false

--------------------------------------------------------
  security.cfg — настройки безопасности
--------------------------------------------------------

[brute_force]

max_attempts
  Максимальное число неудачных попыток входа до блокировки.
  Тип: целое число
  По умолчанию: 5

lockout_minutes
  Время блокировки аккаунта после превышения max_attempts (минуты).
  Тип: целое число
  По умолчанию: 15

attempt_window_minutes
  Окно времени, в котором считаются попытки (минуты).
  Тип: целое число
  По умолчанию: 10

[password]

min_length
  Минимальная длина пароля.
  Тип: целое число
  По умолчанию: 8

require_uppercase
  Требовать хотя бы одну заглавную букву.
  Допустимые значения: true / false

require_lowercase
  Требовать хотя бы одну строчную букву.
  Допустимые значения: true / false

require_digits
  Требовать хотя бы одну цифру.
  Допустимые значения: true / false

require_special
  Требовать хотя бы один специальный символ из special_chars.
  Допустимые значения: true / false

special_chars
  Строка допустимых специальных символов.
  По умолчанию: !@#$%^&*()_+-=[]{}|;:,.<>?

[api_signing]

enabled
  Проверять HMAC-подпись входящих API-запросов.
  Допустимые значения: true / false
  По умолчанию: true

timestamp_tolerance_seconds
  Допустимое расхождение timestamp в запросе (защита от replay-атак).
  Тип: целое число (секунды)
  По умолчанию: 30

--------------------------------------------------------
  logging.cfg — логирование
--------------------------------------------------------

[logging]

level
  Уровень логирования приложения.
  Допустимые значения: DEBUG, INFO, WARNING, ERROR, CRITICAL
  По умолчанию: INFO

file
  Путь к файлу лога приложения.
  Пример: logs/app.log
  Если директория не существует — будет создана автоматически.

max_bytes
  Максимальный размер файла лога до ротации (байты).
  По умолчанию: 10485760 (10 МБ)

backup_count
  Число хранимых архивных файлов лога.
  По умолчанию: 5

format
  Формат строки лога (Python logging format).
  По умолчанию: %(asctime)s | %(levelname)s | %(name)s | %(message)s

[audit]

enabled
  Вести отдельный лог аудита действий (вход, выдача/сброс лицензий и т.д.).
  Допустимые значения: true / false
  По умолчанию: true

file
  Путь к файлу аудит-лога.
  Пример: logs/audit.log

db_enabled
  Дополнительно записывать аудит-события в таблицу license_actions БД.
  Допустимые значения: true / false
  По умолчанию: true

========================================================
  Переменные окружения (приоритет над .cfg)
========================================================

  SECRET_KEY       — app.secret_key
  API_SECRET       — app.api_secret
  DATABASE_URL     — database.url
  SMTP_USER        — smtp.user
  SMTP_PASSWORD    — smtp.password
